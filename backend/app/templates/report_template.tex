% 表格渲染宏
{% macro render_table(block) -%}
\begin{table}[H]
    \centering
    {% if block.caption %}\caption{ {{ block.caption | latex_escape }} }{% endif %}
    {% set cols = 'c' * (block.headers|length) %}
    \begin{tabular}{@{}{{ cols }}@{}}
        \toprule
        {%- for header in block.headers -%}
            {{ header | latex_escape }}{% if not loop.last %} & {% endif %}
        {%- endfor %} \\
        \midrule
        {%- for row in block.rows -%}
            {%- for item in row -%}
                {% if item is string %}{{ item | latex_escape }}{% else %}{{ '{:.4g}'.format(item) }}{% endif %}{% if not loop.last %} & {% endif %}
            {%- endfor %} \\
        {%- endfor %}
        \bottomrule
    \end{tabular}
\end{table}
{%- endmacro %}
